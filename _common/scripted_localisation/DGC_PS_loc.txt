defined_text = { # Country
	name = GetNameSHXprogress
	text = {
		trigger = {
			SHX = { exist = yes }
			SHX = { has_completed_focus = CHI_join_the_republican_government }
			NOT = { has_country_flag = china_leader }
			NOT = { has_country_flag = out_of_PS }
		}
		localization_key = SHXprogress_number
	}
	text = {
		trigger = {
			OR = {
				SHX = { exist = no }
				SHX = { has_country_flag = out_of_PS }
			}
		}
		localization_key = SHXprogress_not_exist
	}
	text = {
		trigger = {
			SHX = { has_country_flag = china_leader }
		}
		localization_key = SHXprogress_leader
	}
}
